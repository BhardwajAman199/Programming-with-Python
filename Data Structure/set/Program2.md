# Addition of elements in a Set 

## Creating a Set 
set1 = set() 
print("Intial blank Set: ",set1) 

##  Adding element and tuple to the Set 
set1.add(1) 
set1.add(2) 
set1.add((3,4)) 
print("\nSet after Addition of Three elements: ",set1)
